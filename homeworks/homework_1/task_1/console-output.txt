scala> var helloMessage: String = "Hello, Scala! This is Dimo4ka"
var helloMessage: String = Hello, Scala! This is Dimo4ka

scala> print(helloMessage)
Hello, Scala! This is Dimo4ka~

scala> var spanishMessage = helloMessage.replaceFirst("Hello", "Hola")

scala> print(spanishMessage)

scala> print(frenchMessage)
